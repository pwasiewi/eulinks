mkdir linux tmp
ls /var/tmp/git/metro/etc/builds
#funtoo-current  funtoo-experimental  funtoo-stable  gentoo-stable
#gentoo-unstable
vim /var/tmp/git/metro/etc/builds/gentoo-unstable/build.conf 
install -d /var/tmp/metro/cache/cloned-repositories
grep name /var/tmp/git/metro/etc/builds/gentoo-unstable/build.conf 
#name: portage
install -d /var/tmp/git/linux/funtoo-current/x86-32bit/core2_64
cd /var/tmp/git/linux/gentoo-stable/x86-64bit/generic_64
install -d 2011-08-22
cd  2011-08-22
wget
 * http://ftp.osuosl.org/pub/funtoo/gentoo-stable/x86-64bit/generic_64/2011-08-22/stage3-generic_64-gentoo-stable-2011-08-22.tar.xz
cd ..
install -d .control/version
echo "2011-08-22" > .control/version/stage3
install -d .control/strategy
echo "local" > .control/strategy/build
echo "stage3" > .control/strategy/seed
cd ..
mkdir core2_64
gen2db x86-64bit # cd core2_64/
install -d .control/strategy
echo "remote" > .control/strategy/build
echo "stage3" > .control/strategy/seed
install -d .control/remote
echo "gentoo-stable" > .control/remote/build
echo "generic_64" > .control/remote/subarch

cd /var/tmp/git/metro
scripts/ezbuild.sh gentoo-stable core2_64
