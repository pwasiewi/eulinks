```
cd /usr/share/minetest/games/minetest_game/mods;
[ ! -d ethereal ] && git clone https://github.com/tenplus1/ethereal
[ ! -d worldedit ] && git clone https://github.com/Uberi/Minetest-WorldEdit.git; 

[ -d Minetest-WorldEdit ] && mv Minetest-WorldEdit worldedit;

#more blocks there
for i in moreblocks moreores moretrees lightning castle torches plantlife_modpack more_chests round_trunks biome_lib unifieddyes; do [ ! -d $i ] && git clone https://github.com/minetest-mods/$i; done;

#more mods there
for i in animalmaterials mobf_core adv_spawning animals_modpack; do [ ! -d $i ] && git clone https://github.com/sapier/$i; done;

#more mods there
for i in mobs_redo mobs_animal mobs_npc; do [ ! -d $i ] && git clone https://github.com/tenplus1/$i; done;

kpgmobs https://forum.minetest.net/download/file.php?id=384;

[ ! -d mobs_scifi ] && git clone https://github.com/D00Med/mobs_scifi;

[ ! -d cme ] && git clone https://github.com/BlockMen/cme;

//find /etc/init.d -type f -exec sed -i 's/runscript/openrc-run/' {} +

//https://github.com/tacigar/maidroid
//https://github.com/bell07/minetest-townchest.git
//https://github.com/eidy/aliveai.git
//https://forum.minetest.net/viewtopic.php?f=9&t=11972

cd /usr/share/minetest/games
git clone http://repo.or.cz/MineClone/MineClone2.git
cd /usr/share/minetest/games/minetest_game/mods
[ ! -d unifieddyes ] && git clone https://github.com/minetest-mods/unifieddyes
[ ! -d castle ] && git clone https://github.com/minetest-mods/castle
[ ! -d minetest-mod-mesecons ] && git clone https://github.com/Jeija/minetest-mod-mesecons.git
[ ! -d mesecons ] && mv minetest-mod-mesecons/* . && rm -rf minetest-mod-mesecons
[ ! -d minetest_orienteering ] && git clone http://repo.or.cz/minetest_orienteering.git
[ ! -d orienteering ] && mv minetest_orienteering orienteering
[ ! -d meshnode ] && git clone https://github.com/stujones11/meshnode.git
[ ! -f ByUjEdwinV2_space.zip ] && wget https://forum.minetest.net/download/file.php?id=7985 -O ByUjEdwinV2_space.zip
[ ! -f ByAiTechEye-juli-V7_1-marssurvive.zip ] && wget https://forum.minetest.net/download/file.php?id=9106 -O ByAiTechEye-juli-V7_1-marssurvive.zip
[ ! -f ByUjEdwinV4r-smartshop.zip ] && wget https://forum.minetest.net/download/file.php?id=8630 -O ByUjEdwinV4r-smartshop.zip
[ ! -f ByAiTechEyeV3_5-aliveai.zip ] && wget https://forum.minetest.net/download/file.php?id=9323 -O ByAiTechEyeV3_5-aliveai.zip
[ ! -f ByAiTechEyeV3_6-aliveai.zip ] && wget https://forum.minetest.net/download/file.php?id=9340 -O ByAiTechEyeV3_6-aliveai.zip
[ ! -f ByAiTechEyeV3_941-aliveai.zip ] && wget https://forum.minetest.net/download/file.php?id=9452 -O ByAiTechEyeV3_941-aliveai.zip
[ ! -d smartshop ] && unzip ByUjEdwinV4r-smartshop.zip
[ ! -d aliveai ] && unzip ByAiTechEyeV3_941-aliveai.zip
mv aliveai/ aliveai.arch
mv aliveai.arch/* .
rm -rf aliveai.arch
sed -i 's/status=false/status=true/g' aliveai/init.lua 
```
